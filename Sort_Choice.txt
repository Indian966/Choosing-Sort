        size = len(unsorted) #정렬크기
        for i in range(size, -1, -1) : heapify(unsorted, size, i)
        for i in range(size-1, -1, -1) :
            unsorted[i], unsorted[0] = unsorted[0], unsorted[i]
            heapify(unsorted, i, 0)

def heapify(unsorted, size, i):
        largest = i
        L = 2 * i + 1
        R = 2 * i + 2
        if L < size and unsorted[L] > unsorted[i]:
                largest = L
        if R < size and unsorted[R] > unsorted[largest]:
                largest = R
        if largest != i:
                unsorted[largest], unsorted[i] = unsorted[i], unsorted[largest]
                heapify(unsorted, size, largest)
